config.vm.define "consul-3" do |consul|
    consul.vm.box = "centos6.5"
    consul.vm.host_name = "consul-3.linbynd.local"
    consul.vm.network "public_network", ip: "192.168.1.72"
end

    consul.vm.box = "centos6.5"
    consul.vm.host_name = "consul-client.linbynd.local"
    consul.vm.network "public_network", ip: "192.168.1.73"
end

    config.vbguest.auto_update = false
end
end
